# pstCreator
Create the PST file for PEST calibration from a variety of inputs

# Usage

```bash
usage: pstCreator.py [-h] [--set_ctl_xls SET_CTL_XLS] [--set_ctl_csv SET_CTL_CSV] [--add_obs_xls ADD_OBS_XLS]
                     [--add_par_xls ADD_PAR_XLS] [--add_pargp_xls ADD_PARGP_XLS] [--add_tied_xls ADD_TIED_XLS]
                     [--add_prior_xls ADD_PRIOR_XLS] [--add_io_xls ADD_IO_XLS] [--add_pp_xls ADD_PP_XLS]
                     [--add_obs_csv ADD_OBS_CSV] [--add_par_csv ADD_PAR_CSV] [--add_pargp_csv ADD_PARGP_CSV]
                     [--add_tied_csv ADD_TIED_CSV] [--add_prior_csv ADD_PRIOR_CSV] [--add_io_csv ADD_IO_CSV]
                     [--add_pp_csv ADD_PP_CSV]
                     pstfile mode

positional arguments:
  pstfile
  mode

options:
  -h, --help            show this help message and exit
  --set_ctl_xls SET_CTL_XLS
  --set_ctl_csv SET_CTL_CSV
  --add_obs_xls ADD_OBS_XLS
  --add_par_xls ADD_PAR_XLS
  --add_pargp_xls ADD_PARGP_XLS
  --add_tied_xls ADD_TIED_XLS
  --add_prior_xls ADD_PRIOR_XLS
  --add_io_xls ADD_IO_XLS
  --add_pp_xls ADD_PP_XLS
  --add_obs_csv ADD_OBS_CSV
  --add_par_csv ADD_PAR_CSV
  --add_pargp_csv ADD_PARGP_CSV
  --add_tied_csv ADD_TIED_CSV
  --add_prior_csv ADD_PRIOR_CSV
  --add_io_csv ADD_IO_CSV
  --add_pp_csv ADD_PP_CSV
```
